//create a file in the folder and (in this example the file is the following)
echo. > bob.txt

//write something in it

//open openssl
openssl

//generate a private key:
genpkey -algorithm RSA -pkeyopt rsa_keygen_bits:2048 -out privatekey.pem

//derive the public key:
pkey -in privatekey.pem -out publickey.pem -pubout

//sign the message with the private key previously generated
dgst -sha256 -sign privatekey.pem -out sign.sha256 bob.txt

	//to get a readable (if base64) version of this file, the follow-up command is
	openssl enc -base64 -in sign.sha256 -out sign.sha256.base64

//now verify the digital signature with the public key
	//first decode the base64 signature
	enc -base64 -d -in sign.sha256.base64 -out sign.sha256

//now veirfy the signature
dgst -sha256 -verify publickey.pem -signature sign.sha256 bob.txt

//you should get the following message
Verified OK

//now, for exercise, modify the message of bob.txt
//then, verify again
dgst -sha256 -verify publickey.pem -signature sign.sha256 bob.txt

//now you get the following message
Verification Failure
error in dgst
